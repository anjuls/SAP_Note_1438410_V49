SELECT
  DU.USERNAME,
  TO_CHAR(COUNT(*), 999999990) NUM_TABLES
FROM
  DBA_USERS DU,
  DBA_TABLES DT
WHERE
  DU.USERNAME = DT.OWNER (+) AND
  DU.USERNAME NOT IN ('SYS', 'SYSTEM', 'DBSNMP', 'SYSMAN', 'TSMSYS', 
    'OUTLN', 'DIP', 'ORACLE_OCM', 'MGMT_VIEW') AND
  DU.USERNAME NOT LIKE 'OPS$%' AND
  DU.USERNAME NOT LIKE 'SAP%'
GROUP BY
  DU.USERNAME
ORDER BY
  DU.USERNAME;
