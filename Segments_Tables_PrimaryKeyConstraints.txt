SELECT NULL OWNER, NULL TABLE_NAME, NULL CONSTRAINT_NAME, NULL IOT_TYPE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL CONSTRAINT_NAME, NULL IOT_TYPE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP___' OWNER
  FROM
    DUAL
)
SELECT
  C.OWNER,
  C.TABLE_NAME,
  C.CONSTRAINT_NAME,
  T.IOT_TYPE
FROM
  BASIS_INFO BI,
  DBA_CONSTRAINTS C,
  DBA_TABLES T
WHERE
  C.OWNER = T.OWNER AND
  C.TABLE_NAME = T.TABLE_NAME AND
  CONSTRAINT_TYPE = 'P' AND
  C.OWNER LIKE BI.OWNER
ORDER BY
  C.OWNER,
  C.TABLE_NAME
));
